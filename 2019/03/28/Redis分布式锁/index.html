#####问题1:如果releaseLock失败，就会导致死锁，所以要给锁加上超时时间

public boolean lock(String lockName,int timeout){
    String lockKey = RedisKey.generate("redis", "lock", lockName);
    Long setnx = redisService.setnx(lockKey, LOCK_FLAG);
    if (1 == setnx){
        redisService.expire(lockKey,timeout);
        return true;
    }else{
        return false;
    }
}

#####问题2：超时问题，锁的时间一般设置比较短，如果业务代码执行时间超过了锁的时间就会出问题，第一个线程没执行完，锁就释放了。第二个线程就会抢到这把锁，等第一个线程执行完成，释放锁，这是第三个线程又会拿到这把锁

//添加lockId 

public Long lock(String lockName, int timeout) {
    String lockKey = RedisKey.generate("redis", "lock", lockName);


    Long lockId = System.currentTimeMillis() + RandomUtils.nextInt();

    Long setnx = redisService.setnx(lockKey, lockId.toString());

    if (1 == setnx) {
        redisService.expire(lockKey, timeout);
        return lockId;
    } else {
        return null;
    }
}


public boolean releaseLock(String lockName, Long lockId) {
    if (StringUtils.isBlank(lockName) || lockId == null){
        return false;
    }
    String lockKey = RedisKey.generate("redis", "lock", lockName);
    try {
        if (lockId.equals(redisService.get(lockKey))){
            redisService.remove(lockKey);
        }
    } catch (Exception e) {
        logger.error("[RedisDistributedLock][releaseLock] 异常！参数: " + lockName + "\n" + e.getMessage(), e);
        return false;
    }
    return true;

}

#####问题6：释放锁的非原子操作

 public static boolean releaseLock(String lockName, String lockId) {
    if (StringUtils.isBlank(lockName) || lockId == null) {
        return false;
    }
    boolean releaseFlag = false;

    String lockKey = RedisKey.generate("redis", "lock", lockName);

    Jedis jedis = null;
    while (true) {
        try {
            jedis = JedisUtil.getJedis(lockKey);
            jedis.watch(lockKey);
            // 判断是不是可以释放锁
            if (lockId.equals(jedis.get(lockKey))) {
                Transaction transaction = jedis.multi();
                Response<Long> del = transaction.del(lockKey);
                transaction.exec();
                if (del.get() != 1) {
                    TimeUnit.MILLISECONDS.sleep(200);
                    continue;
                }
                releaseFlag = true;
            }
            jedis.unwatch();
            break;
        } catch (Exception e) {
            //删除锁失败或者
            try {
                TimeUnit.SECONDS.sleep(1);
            } catch (InterruptedException e1) {
                Thread.currentThread().interrupt();
            }
            //长时间抛这个异常，需要人工介入。。
            logger.error("[RedisDistributedLock][releaseLock] 异常！参数: " + lockName + "\n" + e.getMessage(), e);
        }finally {
            if (jedis != null){
                jedis.close();
            }
        }
    }
    return releaseFlag;
}
